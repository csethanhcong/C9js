Display Logarithmic Data as Bar Chart

# logarithmic_data_bar_chart.js

Edit 
var data1 = [
    {name: 'Monaco', value: 37863},
    {name: 'Liechtenstein', value: 37776},
    {name: 'San Marino', value: 31950},
    {name: 'Greenland', value: 56196},
    {name: 'Viet Nam', value: 94444200},
    {name: 'Egypt', value: 93383574},
    {name: 'Germany', value: 80682351},
    {name: 'U.S.', value: 324118787},
    {name: 'Brazil', value: 209567920},
    {name: 'Palau', value: 21501},
];

var options1 = {
    id: "demo1",
    width: 1100, 
    height: 500,
    data: {
        plain: data1,
    },
    title: {
        text: "The Populations in some Countries"
    },
    isLogaric: true,
    axis: {
        y: {
            tick: {
                format: d3.format(',.1s')
            }
        }
    }
};

var barChart = new C9.BarChart(options1);
barChart.draw();
