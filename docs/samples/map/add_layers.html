var options1 = {
    id: "demo1",
    layers: [
        {
            type: 'Tile',
            source: {
                name: 'BingMaps',
                key: 'your Bing Maps key from http://www.bingmapsportal.com/ here'
            }
        }, {
            type: 'Vector',
            source: {
                name: 'Vector',
                format: 'GeoJSON',
                url: 'https://openlayers.org/en/v3.19.1/examples/data/geojson/countries.geojson'
            }
        }
    ]
};

var map = new C9.Map(options1);
